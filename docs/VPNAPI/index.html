    BatchDeleteVpn Delete /openapi/v1/{omadacId}/sites/{siteId}/vpn Batch delete VPN
    BatchDeleteVpnUser Delete /openapi/v1/{omadacId}/sites/{siteId}/vpn/users Batch delete VPN user
    CheckUsedInVpns Post /openapi/v1/{omadacId}/sites/{siteId}/vpn/checkUsed Check used function for multiple VPN items
    CheckUsedVpn Get /openapi/v1/{omadacId}/sites/{siteId}/vpn/{vpnId}/checkUsed Check used function for single VPN item
    CheckValueAvailable Post /openapi/v1/{omadacId}/sites/{siteId}/vpn/checkValue Check whether vpn value is available
    CreateClientToSiteVpnClient Post /openapi/v1/{omadacId}/sites/{siteId}/vpn/client-to-site-vpn-clients Create client-to-site VPN client
    CreateClientToSiteVpnServer Post /openapi/v1/{omadacId}/sites/{siteId}/vpn/client-to-site-vpn-servers Create client-to-site VPN server
    CreateIpsecFailover Post /openapi/v1/{omadacId}/sites/{siteId}/vpn/ipsec_failovers Create IPsec failover
    CreateS2SAutoVpn Post /openapi/v2/{omadacId}/sites/{siteId}/vpn/site-to-site-vpns/auto Create site-to-site VPN by auto
    CreateS2SManualVpn Post /openapi/v2/{omadacId}/sites/{siteId}/vpn/site-to-site-vpns Create site-to-site VPN by manual
    CreateSiteToSiteVpn Post /openapi/v1/{omadacId}/sites/{siteId}/vpn/site-to-site-vpns Create site-to-site VPN
    CreateVpnClient Post /openapi/v2/{omadacId}/sites/{siteId}/vpn/client-to-site-vpn-clients Create client-to-site VPN client V2
    CreateVpnServer Post /openapi/v2/{omadacId}/sites/{siteId}/vpn/client-to-site-vpn-servers Create client-to-site VPN server V2
    CreateVpnUser Post /openapi/v1/{omadacId}/sites/{siteId}/vpn/users Create VPN user
    CreateVpnUserV2 Post /openapi/v2/{omadacId}/sites/{siteId}/vpn/users Create VPN user V2
    CreateVpnUserV3 Post /openapi/v3/{omadacId}/sites/{siteId}/vpn/users Create VPN user V3
    DeleteClientToSiteVpnClient Delete /openapi/v1/{omadacId}/sites/{siteId}/vpn/client-to-site-vpn-clients/{vpnId} Delete client-to-site VPN client
    DeleteClientToSiteVpnServer Delete /openapi/v1/{omadacId}/sites/{siteId}/vpn/client-to-site-vpn-servers/{vpnId} Delete client-to-site VPN server
    DeleteIpsecFailover Delete /openapi/v1/{omadacId}/sites/{siteId}/vpn/ipsec_failovers/{failoverId} Delete IPsec failover
    DeleteSiteToSiteVpn Delete /openapi/v1/{omadacId}/sites/{siteId}/vpn/site-to-site-vpns/{vpnId} Delete site-to-site VPN
